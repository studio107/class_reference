public function find($templatePath)
{
    $templates = [];
    foreach ($this->finders as $finder) {
        $template = $finder->find($templatePath);
        if ($template !== null) {
            $templates[] = $template;
        }
    }

    // TODO log $templates

    return array_shift($templates);
}

public function getPaths()
{
    $paths = [];
    foreach ($this->finders as $finder) {
        $paths = array_merge($paths, $finder->getPaths());
    }

    return $paths;
}
