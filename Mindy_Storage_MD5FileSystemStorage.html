public function getValidFileName($name)
{
    $fileName = pathinfo($name, PATHINFO_FILENAME);
    $ext = pathinfo($name, PATHINFO_EXTENSION);
    return md5(str_replace("." . $ext, "", $fileName)) . "." . $ext;
}
