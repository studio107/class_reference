public function getAvailableName($name)
{
    $ext = pathinfo($name, PATHINFO_EXTENSION);
    $fileName = str_replace("." . $ext, "", $name);

    $count = 0;
    while ($this->exists($name)) {
        $count += 1;
        $name = strtr("{filename}_{count}.{ext}", [
            '{filename}' => $fileName,
            '{count}' => $count,
            '{ext}' => $ext
        ]);
    }
    return $name;
}

Returns a filename that's free on the target storage system, and available for new content to be written to.
