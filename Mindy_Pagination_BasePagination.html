    $uri = parse_url($_SERVER['REQUEST_URI']);
    if (!isset($uri['query'])) {
        $uri['query'] = '';
    }
    parse_str($uri['query'], $params);
    $params[$this->getPageSizeKey()] = $pageSize;
    return "?" . http_build_query($params);
}
