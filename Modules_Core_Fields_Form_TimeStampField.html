public function getValue(){
    $value = parent::getValue();

    if (!$value && $this->autoNow) {
        $this->value = $value = time();
    }

    if(is_numeric($value) === false) {
        $value = strtotime($value);
    }

    return date($this->format, $value ? $value : time());
}
