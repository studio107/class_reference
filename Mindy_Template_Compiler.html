public function raw($raw, $indent = 0)
{
    $this->line = $this->line + substr_count($raw, "\n");
    if (!fwrite($this->fp, str_repeat(' ', 4 * $indent) . $raw)) {
        throw new RuntimeException('failed writing to file: ' .  $this->target);
    }

    return $this;
}
