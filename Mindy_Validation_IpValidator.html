public function validate($value)
{
    if (
        empty($value) ||
        $this->version == 4 && !preg_match($this->v4pattern, $value) ||
        $this->version == 6 && !preg_match($this->v6pattern, $value)
    ) {
        $this->addError(Translate::getInstance()->t('validator', 'Is not a valid IP address.'));
    }

    return $this->hasErrors() === false;
}
