public function setValue($value)
{
    if (is_object($value)) {
        $this->setOldValue();
        $this->value = $value->getUrl();
    } elseif ($value == $this->cleanValue || is_null($value)) {
        $this->setOldValue();
        $this->value = null;
    } elseif (is_string($value) || is_array($value)) {
        $this->setOldValue();
        $this->value = $value;
    }
    return $this;
}
