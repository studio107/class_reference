public function renderInternal($_viewFile_, $_data_ = null)
{
    // we use special variable names here to avoid conflict when extracting data
    if (is_array($_data_)) {
        extract($_data_, EXTR_PREFIX_SAME, 'data');
    } else {
        $data = $_data_;
    }

    ob_start();
    ob_implicit_flush(false);
    require($_viewFile_);
    return ob_get_clean();
}
