public function write($session_id, $session_data)
{
    // exception must be caught in session write handler
    // http://us.php.net/manual/en/function.session-set-save-handler.php
    $expire = time() + (int)ini_get('session.gc_maxlifetime');

    if($this->encrypt) {
        $session_data = mcrypt_encrypt(MCRYPT_3DES, $this->key, $session_data, MCRYPT_MODE_ECB);
    }

    $session = Session::objects()->filter(['id' => $session_id])->get();
    if ($session === null) {
        $session = new Session([
            'id' => $session_id,
            'data' => $session_data,
            'expire' => $expire,
        ]);
        $saved = $session->save();
        if ($saved === false) {
            throw new Exception("Can't create session");
        }
    } else {
        Session::objects()->filter(['id' => $session_id])->update([
            'data' => $session_data,
            'expire' => $expire
        ]);
    }
    return true;
}

Session write handler. Do not call this method directly.
