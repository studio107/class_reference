public function beginTransaction()
{
    if (!$this->nestable() || $this->transLevel == 0) {
        parent::beginTransaction();
    } else {
        $this->exec("SAVEPOINT LEVEL{$this->transLevel}");
    }

    $this->transLevel++;
}
