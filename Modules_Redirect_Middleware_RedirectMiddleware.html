public function processRequest(Request $request)
{
    $pathInfo = $request->path;
    if (!empty($pathInfo)) {
        $model = Redirect::objects()->filter(['from_url' => '/' . ltrim($pathInfo, '/')])->get();
        if($model === null) {
            return false;
        } else {
            $request->redirect($model->to_url, $model->type);
        }
    }
    return false;
}
